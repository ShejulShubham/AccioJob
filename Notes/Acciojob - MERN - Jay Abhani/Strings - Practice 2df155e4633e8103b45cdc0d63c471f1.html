✅ 1. Count vowels in a string

💡 Problem:

Given a string, count how many vowels (a, e, i, o, u) it contains.

🔧 Solution:

function countVowels(str) {
  // Convert to lowercase to handle capital letters
  str = str.toLowerCase();

  // Set of vowels for quick lookup
  let vowels = new Set(['a', 'e', 'i', 'o', 'u']);
  let count = 0;

  // Loop through each character
  for (let char of str) {
    if (vowels.has(char)) {
      count++;
    }
  }

  return count;
}

// Example
console.log(countVowels("Ujjwal Patel")); // Output: 4

⏱️ Time Complexity: O(n)

🧠 Space Complexity: O(1)

✅ 2. Reverse a string

💡 Problem:

Given a string, return its reverse.

🔧 Solution:

function reverseString(str) {
  // Split string into array, reverse it, and join back
  return str.split("").reverse().join("");
}

// Example
console.log(reverseString("JavaScript")); // Output: "tpircSavaJ"

⏱️ Time Complexity: O(n)

🧠 Space Complexity: O(n) (due to split array)

✅ 3. Check if a string is a palindrome

💡 Problem:

Return true if the string reads the same forward and backward.

🔧 Solution:

function isPalindrome(str) {
  // Remove spaces and convert to lowercase
  str = str.toLowerCase().replace(/\s/g, "");

  // Compare original string with reversed version
  let reversed = str.split("").reverse().join("");
  return str === reversed;
}

// Example
console.log(isPalindrome("madam"));       // true
console.log(isPalindrome("nurses run"));  // true

⏱️ Time Complexity: O(n)

🧠 Space Complexity: O(n)

✅ 4. Replace all spaces with dashes

💡 Problem:

Given a sentence, replace every space (" ") with a dash ("-").

🔧 Solution:

function replaceSpaces(str) {
  return str.replace(/ /g, "-");
}

// Example
console.log(replaceSpaces("Web Development is fun"));
// Output: "Web-Development-is-fun"

⏱️ Time Complexity: O(n)

🧠 Space Complexity: O(n)

✅ 5. Count words in a sentence

💡 Problem:

Count the number of words in a given sentence.

🔧 Solution:

function countWords(sentence) {
  // Trim to remove leading/trailing spaces and split by space
  let words = sentence.trim().split(/\s+/); // Handles multiple spaces
  return words.length;
}

// Example
console.log(countWords("   I love JavaScript  ")); // Output: 3

⏱️ Time Complexity: O(n)

🧠 Space Complexity: O(n)
