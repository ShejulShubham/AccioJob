✅ Problem 1: Sum of All Elements

📝 Problem:

Given an array of numbers, return the sum of all elements.

💡 Example:

Input: [1, 2, 3, 4]

Output: 10

✅ Solution:

function sumArray(arr) {
  let sum = 0;
  for (let num of arr) {
    sum += num;
  }
  return sum;
}

console.log(sumArray([1, 2, 3, 4])); // Output: 10

✅ Problem 2: Find Maximum Element

📝 Problem:

Return the largest number in the array.

💡 Example:

Input: [10, 22, 5, 7]

Output: 22

✅ Solution:

function findMax(arr) {
  let max = arr[0];
  for (let num of arr) {
    if (num > max) {
      max = num;
    }
  }
  return max;
}

console.log(findMax([10, 22, 5, 7])); // Output: 22

✅ Problem 3: Count Even Numbers

📝 Problem:

Count how many even numbers are present in the array.

💡 Example:

Input: [1, 4, 6, 7, 8]

Output: 3

✅ Solution:

function countEvens(arr) {
  let count = 0;
  for (let num of arr) {

      count++;
    }
  }
  return count;

console.log(countEvens([1, 4, 6, 7, 8])); // Output: 3

✅ Problem 4: Reverse an Array

📝 Problem:

Return a new array with the elements in reverse order.

💡 Example:

Input: [1, 2, 3]

Output: [3, 2, 1]

✅ Solution:

function reverseArray(arr) {
  let reversed = [];
  for (let i = arr.length - 1; i >= 0; i--) {
    reversed.push(arr[i]);
  }
  return reversed;
}

console.log(reverseArray([1, 2, 3])); // Output: [3, 2, 1]

✅ Problem 5: Check if Array Contains Element

📝 Problem:

Write a function to check if a given element exists in an array.

💡 Example:

Input: [5, 10, 15], 10

Output: true

✅ Solution:

function containsElement(arr, element) {
  for (let item of arr) {
    if (item === element) {
      return true;
    }
  }
  return false;
}

console.log(containsElement([5, 10, 15], 10)); // Output: true
