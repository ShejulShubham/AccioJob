Nested Arrays Practice

✅ Problem 1: Access a Value

Given the following nested array:

let data = [[10, 20], [30, 40], [50, 60]];

Question: Print the value 40.

✅ Solution:

console.log(data[1][1]); // Output: 40

Explanation:

  • data[1] is [30, 40]
  • data[1][1] is 40

✅ Problem 2: Change a Value

Given:

let marks = [[70, 80], [90, 100]];

Question: Change the value 90 to 95.

✅ Solution:

marks[1][0] = 95;
console.log(marks); // Output: [[70, 80], [95, 100]]

✅ Problem 3: Print All Values (Row by Row)

Given:

let grid = [[1, 2], [3, 4], [5, 6]];

Question: Print each number one by one using a loop.

✅ Solution:

for (let i = 0; i < grid.length; i++) {
  for (let j = 0; j < grid[i].length; j++) {
    console.log(grid[i][j]);
  }
}

Output:

1
2
3
4
5
6

✅ Problem 4: Count Total Elements

Given a nested array:

let items = [[1, 2, 3], [4, 5], [6]];

Question: Count and print the total number of individual numbers inside the nested array.

✅ Solution:

let count = 0;

for (let i = 0; i < items.length; i++) {
  count += items[i].length;
}

console.log("Total elements:", count); // Output: 6

✅ Problem 5: Sum of All Elements

Given:

let matrix = [[2, 4], [6, 8], [10, 12]];

Question: Find and print the sum of all numbers.

✅ Solution:

let sum = 0;

for (let i = 0; i < matrix.length; i++) {
  for (let j = 0; j < matrix[i].length; j++) {
    sum += matrix[i][j];
  }
}

console.log("Total sum:", sum); // Output: 42

✅ Problem 6: Flatten the Nested Array

Given:

let nested = [[1, 2], [3, 4], [5, 6]];

Question: Convert this into a single flat array: [1, 2, 3, 4, 5, 6]

✅ Solution:

let flat = [];

for (let i = 0; i < nested.length; i++) {
  for (let j = 0; j < nested[i].length; j++) {
    flat.push(nested[i][j]);
  }
}

console.log(flat); // Output: [1, 2, 3, 4, 5, 6]

Alternatively, using .flat() method:

console.log(nested.flat()); // Output: [1, 2, 3, 4, 5, 6]

✅ Problem 8: Replace All Even Numbers with 0

Given:

let data = [[1, 2], [3, 4], [5, 6]];

Question: Replace all even numbers with 0.

✅ Solution:

for (let i = 0; i < data.length; i++) {
  for (let j = 0; j < data[i].length; j++) {
    if (data[i][j] % 2 === 0) {
      data[i][j] = 0;
    }
  }
}

console.log(data); // Output: [[1, 0], [3, 0], [5, 0]]

✅ Problem 9: Count How Many Odd Numbers Are There

Given:

let nums = [[2, 5], [7, 8], [9, 12]];

Question: Count how many odd numbers are in the array.

✅ Solution:

let count = 0;

for (let i = 0; i < nums.length; i++) {
  for (let j = 0; j < nums[i].length; j++) {
    if (nums[i][j] % 2 !== 0) {
      count++;
    }
  }
}

console.log("Odd numbers count:", count); // Output: 3

✅ Problem 11: Reverse Each Inner Array

Given:

let scores = [[1, 2], [3, 4], [5, 6]];

Question: Reverse the order of elements inside each inner array.

✅ Solution:

for (let i = 0; i < scores.length; i++) {
  scores[i].reverse();
}

console.log(scores); // Output: [[2, 1], [4, 3], [6, 5]]

🔹 Part 1: Reverse a 1D Array (Same Array)

Example:

let arr = [10, 20, 30, 40];

✅ Goal: Reverse the array in place (modify original array)

🔁 Using Loop (Swapping from both ends)

let arr = [10, 20, 30, 40];
let n = arr.length;

for (let i = 0; i < n / 2; i++) {
  let temp = arr[i];
  arr[i] = arr[n - 1 - i];
  arr[n - 1 - i] = temp;
}

console.log(arr); // [40, 30, 20, 10]

🔍 Explanation:

  • Start from both ends: index 0 and n-1
  • Swap arr[0] and arr[3], then arr[1] and arr[2], etc.
  • Stop when they meet in the middle

🔹 Part 2: Reverse a 1D Array into a New Array

Example:

let arr = [10, 20, 30, 40];

✅ Goal: Create a new reversed array without modifying the original

🔁 Using Loop:

let reversed = [];

for (let i = arr.length - 1; i >= 0; i--) {
  reversed.push(arr[i]);
}

console.log(reversed); // [40, 30, 20, 10]

🔹 Part 3: Reverse Each Inner Array in a 2D Array (Same Array)

Example:

let matrix = [
  [1, 2, 3],
  [4, 5, 6],
  [7, 8, 9]
];

✅ Goal: Reverse each inner array using loop (not .reverse())

🔁 Solution:

for (let i = 0; i < matrix.length; i++) {
  let row = matrix[i];
  let n = row.length;

  for (let j = 0; j < n / 2; j++) {
    let temp = row[j];
    row[j] = row[n - 1 - j];
    row[n - 1 - j] = temp;
  }
}

console.log(matrix);
// Output:
// [
//   [3, 2, 1],
//   [6, 5, 4],
//   [9, 8, 7]
// ]

🔹 Part 4: Create a New 2D Array with Reversed Inner Arrays

Example:

let matrix = [
  [1, 2],
  [3, 4],
  [5, 6]
];

✅ Goal: Create a new 2D array where each inner array is reversed.

🔁 Solution:

let newMatrix = [];

for (let i = 0; i < matrix.length; i++) {
  let reversedRow = [];
  for (let j = matrix[i].length - 1; j >= 0; j--) {
    reversedRow.push(matrix[i][j]);
  }
  newMatrix.push(reversedRow);
}

console.log(newMatrix);
// Output:
// [
//   [2, 1],
//   [4, 3],
//   [6, 5]
// ]
