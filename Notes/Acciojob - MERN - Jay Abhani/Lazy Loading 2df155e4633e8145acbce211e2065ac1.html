🔹 Dashboard.jsx

🔹 Profile.jsx

  console.log("Profile loaded");
  return <h2>Profile Page</h2>;
}

🔹 App.jsx

import { useState, Suspense } from "react";
import React from "react";

// Lazy loaded components
const Dashboard = React.lazy(() => import("./Dashboard"));
const Profile = React.lazy(() => import("./Profile"));

function App() {
  const [page, setPage] = useState("dashboard");

  return (
    <div>
      <h1>Lazy Loading Demo</h1>

      <button onClick={() => setPage("dashboard")}>
        Dashboard
      </button>

      <button onClick={() => setPage("profile")}>
        Profile
      </button>

      <Suspense fallback={<h2>Loading component...</h2>}>
        {page === "dashboard" && <Dashboard />}
        {page === "profile" && <Profile />}
      </Suspense>
    </div>
  );
}

export default App;

2️⃣ Real App Structure

A real React app looks like this:

src/
 ├── pages/
 │    ├── Home.jsx
 │    ├── Login.jsx
 │    ├── Dashboard.jsx
 │    └── Settings.jsx
 ├── App.jsx
 └── main.jsx

Each page = route = lazy loaded

3️⃣ Core Rule (IMPORTANT)

❌ Don’t lazy load like this

<button />
<Card />
<Input />

✅ Lazy load like this

/Home
/Login
/Dashboard
/Settings

👉 One route = one chunk

5️⃣ Lazy Loading Pages (CORRECT WAY)

App.jsx

import React, { Suspense } from "react";
import { Routes, Route } from "react-router-dom";

// Lazy-loaded pages
const Home = React.lazy(() => import("./pages/Home"));
const Login = React.lazy(() => import("./pages/Login"));
const Dashboard = React.lazy(() => import("./pages/Dashboard"));

function App() {
  return (
    <Suspense fallback={<h2>Loading page...</h2>}>
      <Routes>
        <Route path="/" element={<Home />} />
        <Route path="/login" element={<Login />} />
        <Route path="/dashboard" element={<Dashboard />} />
      </Routes>
    </Suspense>
  );
}

export default App;
