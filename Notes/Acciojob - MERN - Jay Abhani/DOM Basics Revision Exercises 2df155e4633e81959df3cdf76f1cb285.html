Task 1 — Solution

// 1) Basic Selectors
const mainDiv = document.getElementById("main");
console.log("byId #main:", mainDiv);

const titles = document.getElementsByClassName("title");
console.log("byClass .title (first):", titles[0]);

const paragraphs = document.getElementsByTagName("p");
console.log("byTag <p> (first):", paragraphs[0]);

const firstItem = document.querySelector(".item");
console.log("querySelector .item:", firstItem);

const allItems = document.querySelectorAll(".item");
console.log("querySelectorAll .item (NodeList):", allItems);
allItems.forEach((li, i) => console.log(`item[${i}]:`, li.textContent));

Task 2 — Solution

// 2) Node vs Element
console.log("#main childNodes.length:", mainDiv.childNodes.length);
console.log("#main children.length:", mainDiv.children.length);

console.log("firstChild:", mainDiv.firstChild);
console.log("firstElementChild:", mainDiv.firstElementChild);
console.log("lastChild:", mainDiv.lastChild);
console.log("lastElementChild:", mainDiv.lastElementChild);

/*
Explanation:
- childNodes includes text nodes (like whitespace/newlines) AND element nodes.
- children includes ONLY element nodes. Hence, these counts often differ.
*/

Task 3 — Solution

// 3) textContent vs innerHTML
const info = document.getElementById("info");

info.textContent = "<b>Safe?</b>";   // renders literally as <b>Safe?</b>
info.innerHTML   = "<b>Safe?</b>";   // renders bold text

/*
textContent does not parse HTML => safer for user input (prevents XSS).
innerHTML parses HTML => powerful, but unsafe with untrusted input.
*/

Task 7 — Solution (requires the extra <img> and button HTML added)

// 7) Toggle image src + alt
const hero = document.getElementById("hero");
const swap = document.getElementById("swap");

swap.addEventListener("click", () => {
  const current = hero.getAttribute("src");
  if (current.includes("text=A")) {
    hero.setAttribute("src", "https://via.placeholder.com/120x60?text=B");
    hero.setAttribute("alt", "Image B");
  } else {
    hero.setAttribute("src", "https://via.placeholder.com/120x60?text=A");
    hero.setAttribute("alt", "Image A");
  }
});

Task 8 — Solution

// 8) className vs classList
// Replace all classes:
mainDiv.className = "box red"; // overwrites earlier classes completely
