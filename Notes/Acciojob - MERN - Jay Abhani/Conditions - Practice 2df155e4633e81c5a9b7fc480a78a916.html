🧪 Q21: Check if a Number is Prime

Problem: Print whether a number is prime or not.

let num = 7;
let isPrime = true;

if (num < 2) {
  isPrime = false;
} else {
  for (let i = 2; i <= Math.sqrt(num); i++) {
    if (num % i === 0) {
      isPrime = false;
      break;
    }
  }
}

console.log(isPrime ? "Prime Number" : "Not a Prime Number");
